class PatternProgram1
{
	 public static void main(String [] args)
	 {
		 int no = 6;
		 for (int i = 0; i<=no;i++)   
		 {
			 int a = no-i;
			 for(int j=0;j<=i;j++)
			 {
				System.out.print(a++);
			 }
			   System.out.println();
		 }
	 }
}
